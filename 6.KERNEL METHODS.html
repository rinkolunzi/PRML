n = 100
def k1(xs, ys, sigma=1):
    dx = xs - ys
    return  np.exp((-(dx) ** 2) / 2/(sigma ** 2))
def k2(xs, ys, theta=1):
    dx = xs - ys
    return  np.exp(-theta*np.abs(dx))
def mm(x):
    return np.zeros_like(x)

    xs=xs.reshape([n,1])
    xs1=xs.reshape([1,n])
    K1= k1(xs, xs1)

noise_level=1e-4

def kk1(xn, xm):
    xnm=xn@xm.T
    return xnm

def k2(xs, ys, theta=1):
    dx = xs - ys
    return  np.exp(-theta*np.abs(dx))
def fit1(X,t,X_n):
    X1=X.reshape(len(X),2)
    Gram = kk1(X, X1)
    covariance = Gram + np.eye(len(Gram)) * noise_level
    precision = np.linalg.inv(covariance)
    X_n1=X_n.reshape(len(X_n),2)
    K =kk1(X_n1, X)
    a_mean = K @ precision @ t
    return a_mean
